#man pages
#man1

              plfs_flatten_index plfs_map plfs_recover dcon findmesgbuf)

endforeach(MAN1)

              plfs_readdir plfs_utime plfs_chmod 
              plfs_mkdir plfs_readlink plfs_statvfs
              plfs_flush_writes plfs_invalidate_read_cache )
